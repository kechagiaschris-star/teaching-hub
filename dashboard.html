-- Τρέξε αυτό στο SQL Editor του Supabase create table if not exists lessons ( id bigint primary key generated always as identity, student text, date text, subject text, topic text, duration int, fee numeric, payment text, type text, notes text, created_at timestamptz default now() ); create table if not exists schedule ( id bigint primary key generated always as identity, student text, day int, start_time text, duration int, subject text, fee numeric, type text ); create table if not exists homework ( id bigint primary key generated always as identity, type text, student text, description text, priority text, deadline text, done boolean default false, created_at timestamptz default now() ); -- Enable Row Level Security (public access for single user) alter table lessons enable row level security; alter table schedule enable row level security; alter table homework enable row level security; create policy "public_all" on lessons for all using (true) with check (true); create policy "public_all" on schedule for all using (true) with check (true); create policy "public_all" on homework for all using (true) with check (true);
